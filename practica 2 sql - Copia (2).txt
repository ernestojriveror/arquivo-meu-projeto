create table paciente(
	codpac int not null primary key ,	
	nome varchar (45) not null,
	datnasc date,
	sexo varchar (2),
	cpf varchar (14) unique,
	telefone varchar (14)
);

create table medico(
	codmed int not null primary key ,	
	nome varchar (80) not null,
	CRM int,
	telefone varchar (14)
);

create  table agendaConsulta(
	codagencons int not null primary key ,	
	datcons date not null,
	horacons time,
	observacao varchar (255),
	codpac int constraint age_codpac_fk references paciente(codpac),
	codmed int constraint age_codmed_fk references medico(codmed)
);
